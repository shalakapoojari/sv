CREATE TABLE pending_users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100),
    mobile VARCHAR(20),
    city VARCHAR(50),
    state VARCHAR(50),
    email VARCHAR(100),
    status VARCHAR(20) DEFAULT 'pending'
);

CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100),
    email VARCHAR(100),
    password VARCHAR(100),
    role VARCHAR(50),
    login_id VARCHAR(50),
    mobile VARCHAR(20),
    city VARCHAR(50),
    state VARCHAR(50),
    status VARCHAR(20)
);

CREATE TABLE IF NOT EXISTS search_logs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    emp_id VARCHAR(100),
    email VARCHAR(100),
    action VARCHAR(255),
    device VARCHAR(255),
    location VARCHAR(255),
    timestamp DATETIME
);

CREATE TABLE IF NOT EXISTS car_search_logs (
    id INT AUTO_INCREMENT PRIMARY KEY,
    emp_id VARCHAR(100) NOT NULL,
    emp_name VARCHAR(100) NOT NULL,
    chasis_no VARCHAR(100) NOT NULL,
    searched_at DATETIME NOT NULL,
    location VARCHAR(255) NOT NULL,
    device_info VARCHAR(255) NOT NULL,
    suspicious TINYINT DEFAULT 0
);
