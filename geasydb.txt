CREATE TABLE pending_users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100),
    mobile VARCHAR(20),
    city VARCHAR(50),
    state VARCHAR(50),
    email VARCHAR(100),
    status VARCHAR(20) DEFAULT 'pending'
);

CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100),
    email VARCHAR(100),
    password VARCHAR(100),
    role VARCHAR(50),
    login_id VARCHAR(50),
    mobile VARCHAR(20),
    city VARCHAR(50),
    state VARCHAR(50),
    status VARCHAR(20)
);
